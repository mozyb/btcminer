  const lines = text.split("\n");

      title="Bitcoin Mining Profitability Calculator | BTCMiner.online"
      description="Use our free Bitcoin mining calculator to estimate daily, weekly, and monthly BTC earnings. Enter your hashrate, electricity cost and difficulty to calculate mining profits."

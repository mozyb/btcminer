   240   // Create or reset job to 'running'
   241   let jobId: string;
   242   if (existingJob) {
   243     jobId = existingJob.id;
   244     await db.from("daily_reward_jobs")
   245       .update({ status: "running", started_at: new Date().toISOString(), error_message: null })
   246       .eq("id", jobId);
   247   } else {
   248     const { data: newJob, error: createErr } = await db
   249       .from("daily_reward_jobs")
   250       .insert({ run_date: runDate, status: "running", started_at: new Date().toISOString() })
   251       .select("id")
   252       .single();
   253     if (createErr) {
   254       log("ERROR", "job", "Failed to create job row", createErr.message);
   255       return new Response(
   256         JSON.stringify({ success: false, error: "Failed to create job record: " + createErr.message }),
   257         { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }
   258       );
   259     }
   260     jobId = newJob.id;
   261   }
   262   log("INFO", "job", "Job row ready", { jobId, runDate });
   263 
   264   try {
   265     // ── Fetch live network stats ─────────────────────────────────────────
   266     const stats = await fetchNetworkStats();
   267 
   268     // ── Load all active contracts ────────────────────────────────────────
   269     const today = new Date().toISOString().slice(0, 10);
   270     // Include lifetime contracts (expiry_date IS NULL) and non-expired contracts
   271     const { data: contracts, error: contractsErr } = await db
   272       .from("contracts")
   273       .select("id, user_id, hashrate, hashrate_unit, maintenance_fee_rate")
   274       .eq("status", "active")
   275       .lte("start_date", today)
   276       .or(`expiry_date.is.null,expiry_date.gte.${today}`);
   277 
   278     if (contractsErr) throw new Error("Fetch contracts failed: " + contractsErr.message);
   279 
   280     log("INFO", "contracts", "Loaded active contracts", { count: contracts?.length ?? 0 });
   281 
   282     if (!contracts || contracts.length === 0) {
   283       await db.from("daily_reward_jobs").update({
   284         status: "completed",
   285         btc_price: stats.btcPrice,
   286         network_hashrate: stats.networkHashrateEH,
   287         network_difficulty: stats.networkDifficultyT,
   288         block_reward: stats.blockReward,
   289         contracts_processed: 0,
   290         total_btc_credited: 0,
   291         completed_at: new Date().toISOString(),
   292       }).eq("id", jobId);
   293 
   294       return new Response(
   295         JSON.stringify({ success: true, processedUsers: 0, rewardsDistributed: 0, timestamp: new Date().toISOString() }),
   296         { headers: { ...corsHeaders, "Content-Type": "application/json" } }
   297       );
   298     }
   299 
   300     // ── Calculate rewards per contract ───────────────────────────────────
   301     let totalCredited = 0;
   302     let contractsProcessed = 0;
   303     const creditInserts: {
   304       job_id: string;
   305       user_id: string;
   306       contract_id: string;
   307       btc_amount: number;
   308       usd_value: number;
   309       network_hashrate: number;
   310     }[] = [];
   311     const userCredits = new Map<string, number>();
   312 
   313     for (const contract of contracts) {
   314       let hashrateTH = contract.hashrate as number;
   315       const unit = ((contract.hashrate_unit as string) ?? "TH/s").toUpperCase();
   316       if (unit === "PH/S") hashrateTH *= 1000;
   317       else if (unit === "GH/S") hashrateTH /= 1000;
   318       else if (unit === "EH/S") hashrateTH *= 1e6;
   319 
   320       const maintenanceRate = (contract.maintenance_fee_rate as number) ?? 0.0028;
   321       const btcAmount = calcDailyRewardBTC(
   322         hashrateTH,
   323         stats.networkHashrateEH,
   324         stats.btcPrice,
   325         stats.blockReward,
   326         maintenanceRate
   327       );
   328 
   329       if (btcAmount <= 0) {
   330         log("INFO", "reward", `Contract ${contract.id} yields 0 BTC — skipped`);
   331         continue;
   332       }
   333 
   334       const usdValue = btcAmount * stats.btcPrice;
   335       creditInserts.push({
   336         job_id: jobId,
   337         user_id: contract.user_id,
   338         contract_id: contract.id,
   339         btc_amount: btcAmount,
   340         usd_value: usdValue,
   341         network_hashrate: stats.networkHashrateEH,
   342       });
   343 
   344       userCredits.set(contract.user_id, (userCredits.get(contract.user_id) ?? 0) + btcAmount);
   345       totalCredited += btcAmount;
   346       contractsProcessed++;
   347     }
   348 
   349     log("INFO", "reward", "Calculated rewards", { contractsProcessed, totalCredited, uniqueUsers: userCredits.size });
   350 
   351     // ── Bulk insert reward_credits ───────────────────────────────────────
   352     if (creditInserts.length > 0) {
   353       const { error: insertErr } = await db.from("reward_credits").insert(creditInserts);
   354       if (insertErr) throw new Error("Insert reward_credits failed: " + insertErr.message);
   355       log("INFO", "db", `Inserted ${creditInserts.length} reward_credits rows`);
   356     }
   357 
   358     // ── Credit each user's BTC wallet via credit_mining_reward RPC ───────
   359     // Uses the new RPC which inserts a single 'mining_reward' transaction
   360     // (avoids double-transaction bug of calling admin_adjust_balance)
   361     let processedUsers = 0;
   362     for (const [userId, btcAmount] of userCredits) {
   363       const usdValue = btcAmount * stats.btcPrice;
   364 
   365       const { error: creditErr } = await db.rpc("credit_mining_reward", {
   366         p_user_id:   userId,
   367         p_currency:  "BTC",
   368         p_amount:    btcAmount,
   369         p_usd_value: usdValue,
   370         p_note:      `Daily mining reward ${runDate}`,
   371       });
   372 
   373       if (creditErr) {
   374         log("ERROR", "credit", `credit_mining_reward failed for user ${userId}`, creditErr.message);
   375         continue;
   376       }
   377 
   378       // Push in-app notification
   379       const { error: notifErr } = await db.from("notifications").insert({
   380         user_id: userId,
   381         title: "Mining Reward Credited",
   382         message: `${btcAmount.toFixed(8)} BTC ($${usdValue.toFixed(2)}) has been credited to your wallet for ${runDate}.`,
   383         type: "mining",
   384       });
   385       if (notifErr) log("WARN", "notif", `Notification insert failed for ${userId}`, notifErr.message);
   386 
   387       // Send digest email (non-blocking — failure must not abort job)
   388       const userContractCount = creditInserts.filter(c => c.user_id === userId).length;
   389       const { data: profile } = await db
   390         .from("profiles")
   391         .select("email, first_name, username")
   392         .eq("id", userId)
   393         .maybeSingle();
   394 
   395       if (profile?.email) {
   396         sendDigestEmail({
   397           userEmail: profile.email,
   398           firstName: profile.first_name ?? profile.username ?? profile.email.split("@")[0],
   399           btcAmount,
   400           usdValue,
   401           btcPrice: stats.btcPrice,
   402           networkHashrate: stats.networkHashrateEH,
   403           contractCount: userContractCount,
   404           runDate,
   405         }).catch(e => log("ERROR", "email", `Digest email failed for ${userId}`, String(e)));
   406       }
   407 
   408       processedUsers++;
   409       log("INFO", "credit", `Credited ${btcAmount.toFixed(8)} BTC to user ${userId}`);
   410     }
   411 
   412     // ── Update contracts.rewards_generated ──────────────────────────────
   413     for (const credit of creditInserts) {
   414       const { error: incrErr } = await db.rpc("increment_contract_reward", {
   415         p_contract_id: credit.contract_id,
   416         p_amount: credit.btc_amount,
   417       });
   418       if (incrErr) log("WARN", "contract", `increment_contract_reward failed for ${credit.contract_id}`, incrErr.message);
   419     }
   420 
   421     // ── Mark job complete ────────────────────────────────────────────────
   422     await db.from("daily_reward_jobs").update({
   423       status: "completed",
   424       btc_price: stats.btcPrice,
   425       network_hashrate: stats.networkHashrateEH,
   426       network_difficulty: stats.networkDifficultyT,
   427       block_reward: stats.blockReward,
   428       contracts_processed: contractsProcessed,
   429       total_btc_credited: totalCredited,
   430       completed_at: new Date().toISOString(),
   431     }).eq("id", jobId);
   432 
   433     log("INFO", "job", "Job completed successfully", { jobId, processedUsers, contractsProcessed, totalCredited });
   434 
   435     return new Response(
   436       JSON.stringify({
   437         success: true,
   438         processedUsers,
   439         rewardsDistributed: totalCredited,
   440         timestamp: new Date().toISOString(),
   441         // Additional context for admin UI
   442         job_id: jobId,
   443         run_date: runDate,
   444         contracts_processed: contractsProcessed,
   445         btc_price: stats.btcPrice,
   446         network_hashrate_eh: stats.networkHashrateEH,
   447       }),
   448       { headers: { ...corsHeaders, "Content-Type": "application/json" } }
   449     );
   450 
   451   } catch (err) {
   452     const msg = err instanceof Error ? err.message : String(err);
   453     const stack = err instanceof Error ? err.stack : undefined;
   454     log("ERROR", "job", "Job failed with exception", { msg, stack, jobId, runDate });
   455 
   456     await db.from("daily_reward_jobs").update({
   457       status: "failed",
   458       error_message: msg,
   459       completed_at: new Date().toISOString(),
   460     }).eq("id", jobId);
   461 
   462     return new Response(
   463       JSON.stringify({
   464         success: false,
   465         error: msg,
   466         timestamp: new Date().toISOString(),
   467         job_id: jobId,
   468       }),
   469       { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }
   470     );
   471   }
   472 });
